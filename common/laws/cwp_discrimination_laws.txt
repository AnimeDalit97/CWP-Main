# group = this is the law_group a law belongs to
# icon = graphical icon shown in-game
# modifier = {} modifier on country for having adopted this law
law_lgbtq_banned = {
	group = lawgroup_discrimination
	progressiveness = 100
	icon = "gfx/interface/icons/law_icons/slavery_banned.dds"

	on_activate = {
	}
	modifier = {
		political_movement_radicalism_mult = 0.15
	}
	pop_support = {
		value = 0
		add = {
			desc = "POP_SLAVES"
			if = {
				limit = {
					is_pop_type = slaves
				}
				value = 1
			}
		}
		add = {
			desc = "POP_ACADEMICS"
			if = {
				limit = {
					is_pop_type = academics
				}
				value = 0.1
			}
		}
		add = {
			desc = "POP_CLERGYMEN"
			if = {
				limit = {
					is_pop_type = clergymen
				}
				value = 0.05
			}
		}
	}
	revolution_state_weight = {
		value = 1
		if = {
			limit = {
				is_slave_state = yes
			}
			if = {
				limit = {
					owner = {
						has_journal_entry = je_acw_countdown
					}
				}
				multiply = 0
			}
			else = {
				multiply = 0.25
			}
		}
	}
}

law_lgbtq_legal_equality = {
	group = lawgroup_discrimination
	progressiveness = 0
	icon = "gfx/interface/icons/law_icons/slave_trade.dds"
	unlocking_laws = {
		law_womens_suffrage
	}
	on_enact = {

	}
	modifier = {
		political_movement_radicalism_mult = 0.1
		state_migration_pull_mult = 0.15
	}
	possible_political_movements = {
		law_slavery_banned
		law_legacy_slavery
	}
	pop_support = {
		value = 0
		add = {
			desc = "POP_ARISTOCRATS"
			if = {
				limit = {
					is_pop_type = aristocrats
				}
				value = 0.1
			}
		}
	}
	ai_will_do = {
		exists = ruler
		ruler = {
			has_ideology = ideology_slaver
		}
	}
	revolution_state_weight = {
		value = 1
		if = {
			limit = {
				is_slave_state = yes
			}
			multiply = 2
		}
		else = {
			if = {
				limit = {
					owner = {
						has_journal_entry = je_acw_countdown
					}
				}
				multiply = 0
			}
			else = {
				multiply = 0.25
			}
		}
	}
}
