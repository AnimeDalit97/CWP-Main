on_monthly_pulse_country = {
	random_country = {
		set_global_variable = {
			name = var_un_memberlist
			value = 0
		}
	}
	every_country = {
		limit = {
			or = {
				has_variable = un_member
				has_variable = un_founder
			}
		}
		change_global_variable = {
			name = var_un_memberlist
			add = 1
		}
	}
}

on_monthly_pulse_country = {
	effect_corruption_value_check = yes
	corruption_check = yes
}
