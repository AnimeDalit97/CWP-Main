establish_military_defence_establishment = {
	is_shown = {
		this = {
			not = {
				has_variable = defence_establishment_established
				has_building = building_military_defence_establishment
			}
		}
	}

	possible = {
		in_default = no
		or = {
			has_technology_researched = operational_warfare
			
		}
		not = {
			every_scope_state = {
				any_scope_building = {
					limit = {
						is_building_type = building_military_defence_establishment
						is_under_construction = yes
					}
				}
			}
		}
	}

	when_taken = {
		set_variable = defence_establishment_established
		any_scope_state = {
			limit = {
				is_capital = yes
			}
			start_building_construction = building_military_defence_establishment
		}
	}

	ai_chance = {
		base = 0
		modifier = {
			trigger = {
				gold_reserves < 1000000
			}
			add = -20
		}
		
	}
}
