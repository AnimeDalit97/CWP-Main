develop_nuclear_weapons = {
	is_shown = {
		this = {
			has_law = law_type:law_nuclear_weapons
			not = {
				has_variable = developing_nuclear_weapons
			}
		}
	}

	possible = {
		in_default = no
		not = {
			has_law = law_type:law_nuclear_disarmament
			has_law = law_type:law_power_generation
		}
	}

	when_taken = {
		set_variable = developing_nuclear_weapons
		add_journal_entry = {
			type = je_acquire_nuclear_capability
		}
	}

	ai_chance = {
		base = 0
		modifier = {
			trigger = {

			}
			add = 20
		}
	}
}
