ai_raise_defcon = {
	is_shown = {
		this = {
			is_ai = yes
			has_variable = defcon_level
		}
	}

	possible = {
		is_ai = yes
		nor = {
			has_variable = defcon_lowered
			var:defcon_level >= 5
		}
		var:defcon_level < 5
	}

	when_taken = {
		raise_defcon = yes
	}

	ai_chance = { # TODO, does not work
		base = 0
		modifier = { 
			trigger = {
				this = {
					ai_raise_defcon_value > ai_lower_defcon_value
				}  
			}
			add = ai_raise_defcon_value
		}
	}
}

ai_lower_defcon = {
	is_shown = {
		this = {
			is_ai = yes
			has_variable = defcon_level
		}
	}

	possible = {
		is_ai = yes
		nor = {
			has_variable = defcon_lowered
			var:defcon_level <= 1
		}
		var:defcon_level > 1
	}

	when_taken = {
		lower_defcon = yes
	}

	ai_chance = { # TODO, does not work
		base = 0
		modifier = {
			trigger = { 
				this = {
					ai_raise_defcon_value < ai_lower_defcon_value 
				}
			}
			add = ai_lower_defcon_value
		}
	}
}
