ai_raise_defcon = {
	is_shown = {
		this = {
			is_ai = yes
		}
	}

	possible = {
		is_ai = yes
		not = {
			has_variable = defcon_lowered
			var:defcon_level <= 1
		}
		var:defcon_level < 5
	}

	when_taken = {
		raise_defcon = yes
	}

	ai_chance = {
		base = 0
		modifier = {
			trigger = {
				or = {
					is_diplomatic_play_committed_participant = yes
					is_at_war = yes
				}
				
			}
			add = -100
		}
		modifier = {
			trigger = {
				or = {
					is_diplomatic_play_committed_participant = no
					is_at_war = no
				}
			}
			add = 100
		}
		modifier = {
			trigger = {
				has_strategy = ai_strategy_armed_isolationism
				var:defcon_level >= 4
			}
			add = -50
		}
		modifier = {
			trigger = {
				any_neighbouring_state = {
					owner = {
						is_at_war = yes
					}
				}
				var:defcon_level >= 4
			}
			add = -50
		}
		modifier = {
			trigger = {
				any_rival_country = {
					var:defcon_level >= 4
				}
			}
			add = 50
		}
	}
}
