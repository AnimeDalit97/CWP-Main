leave_treaty_of_five_eyes = {
	is_shown = {
		any_country = {
			has_diplomatic_pact = {
				who = ROOT
				type = treaty_five_eyes
			}
		}
	}

	possible = {
		any_country = {
			has_diplomatic_pact = {
				who = ROOT
				type = treaty_five_eyes
			}
		}
	}

	when_taken = {
		every_country = {
			limit = {
				has_diplomatic_pact = {
					who = ROOT
					type = treaty_five_eyes
				}
			}
			change_relations = {
				tcountry = root
				value = -50
			}
			remove_diplomatic_pact = {
				country = ROOT
				type = treaty_five_eyes
			}
		}
	}

	ai_chance = {
		base = 0
		modifier = {
			trigger = {
				gold_reserves < 1000000
			}
			add = -20
		}
		modifier = {
			trigger = {
				country_rank > rank_value:great_power
			}
			add = 100
		}
		modifier = {
			trigger = {
				country_rank > rank_value:superpower
			}
			add = 1000
		}
	}
}
