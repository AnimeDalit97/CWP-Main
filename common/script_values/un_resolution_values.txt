#corruption values

un_support_value = {
	value = global_var:is_supporting_list
	round = yes
}

un_oppose_value = {
	value = global_var:is_opposing_list
	round = yes
}

voting_balance = {
	value = un_support_value
	subtract = un_oppose_value
	round = yes
}

diplomatic_weight = {
	value = 0
	add = root.gdp
	add = root.prestige
	divide = 10000
	round = yes
}

doubleprogressbar_supportive = {
	value = un_support_value
	divide = 100
	min = 0
}

doubleprogressbar_opposing = {
	value = un_oppose_value
	divide = 100
	min = 0
}

doubleprogressbar_dynamic = {
	value = 0
	if = {
		limit = {
			un_support_value > un_oppose_value
		}
		value = un_support_value
	}
	if = {
		limit = {
			un_oppose_value > un_support_value
		}
		value = un_oppose_value
	}
}
