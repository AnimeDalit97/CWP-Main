#corruption values

corruption_value = {
	value = var:var_corruption
	max = corruption_value_max
	min = 0
}

corruption_growth = {
	value = 2
	if = {
		limit = {
			exists = this 
			this = {
				has_law = law_type:law_medium_corruption
			}
		}
		add = 2
	}
}

corruption_decay = {
	value = 1
	if = {
		limit = {
		var:var_corruption <= corruption_baseline
		}
		add = 2
	}
	if = {
		limit = {
			exists = this 
			this = {
				has_law = law_type:law_medium_corruption
			}
		}
		add = 2
	}

	if = {
		limit = {
			exists = this 
			this = {
				has_law = law_type:law_high_corruption
			}
		}
		add = 5
	}
}

corruption_baseline = {
	value = 100
	if = {
		limit = {
			exists = this 
			this = {
				has_law = law_type:law_medium_corruption
			}
		}
		subtract = 20
	}
	if = {
		limit = {
			exists = this 
			this = {
				has_law = law_type:law_high_corruption
			}
		}
		subtract = 40
	}
	max = 100
	min = 0
}

corruption_value_max = {
	value = 100
}

corruption_modifier = {
	value = this
	add = corruption_value
	divide = 3
	max = 100
	min = 0
}
