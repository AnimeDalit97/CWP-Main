# This is executed last among all history
GLOBAL = {
	## UN Variables
	random_country = {
		set_global_variable = {
			name = var_un_memberlist
			value = 0
		}
	}
	every_country = {
		limit = {
			has_variable = un_member
		}
		change_global_variable = {
			name = var_un_memberlist
			add = 1
		}
	}
	## Countries List
	every_country = {
		add_to_global_variable_list = {
			name = countries_list
			target = THIS
		}
	}
	## Corruption Variables
	every_country = {
		limit = {
			not = {
				has_variable = var_corruption
			}
		}
		set_variable = {
			name = var_corruption
			value = 0
		}
	}
	## DEFCON Setup
	every_country = {
		if = {
			limit = {
				not = {
					has_variable = defcon_level
				}
				is_at_war = no
			}
			set_variable = {
				name = defcon_level
				value = 5
			}
		}
	}
	every_country = {
		if = {
			limit = {
				not = {
					has_variable = defcon_level
				}
				is_at_war = yes
			}
			set_variable = {
				name = defcon_level
				value = 2
			}
		}
	}
	## Pollution Variables
	set_global_variable = {
		name = global_var_pollution
		value = global_pollution_value
	}
	set_global_variable = {
		name = global_var_temperature_celsius
		value = global_temperature_value_celsius
	}
	## Dynamic Peace deal State Setup
	effect_china_states_scopes_setup = yes
}
