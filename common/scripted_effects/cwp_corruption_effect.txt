### Corruption Value determiner
effect_corruption_value_check = {
	if = {
		limit = {
			var:var_corruption < 0
		}
		set_variable = {
			name = var_corruption
			value = 0
		}
	}
	if = {
		limit = {
			var:var_corruption >= 100
		}
		set_variable = {
			name = var_corruption
			value = 100
		}
	}
	if = {
		limit = {
			var:var_corruption >= corruption_baseline 
		}
		change_variable = {
			name = var_corruption
			subtract = corruption_decay
		}
	}
	if = {
		limit = {
			var:var_corruption <= corruption_baseline
		}
		change_variable = {
			name = var_corruption
			add = corruption_growth
		}
	}
	else_if = {
		set_variable = {
			name = var_corruption
			value = corruption_value
		}
	}
}
