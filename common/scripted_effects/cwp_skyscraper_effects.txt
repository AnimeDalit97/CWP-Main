set_too_many_skyscrapers = {
	set_local_variable = {
		name = too_many_skyscrapers_mult
		value = $LEVEL$
	}
	change_local_variable = {
		name = too_many_skyscrapers_mult
		multiply = local_var:too_many_skyscrapers_mult
		multiply = 0.33
	}
	add_modifier = {
		name = too_many_skyscrapers
		multiplier = local_var:too_many_skyscrapers_mult
	}
}
