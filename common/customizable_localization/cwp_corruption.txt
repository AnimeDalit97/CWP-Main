corruption_val = {
	type = country
	text = {
		trigger = {
			AND = {
				corruption_value >= 0
				corruption_value <= 39
			}
		}
		localization_key = corruption_low
	}
	text = {
		trigger = {
			AND = {
				corruption_value >= 40
				corruption_value <= 59
			}
		}
		localization_key = corruption_med
	}
	text = {
		trigger = {
			AND = {
				corruption_value >= 60
				corruption_value <= 100
			}
		}
		localization_key = corruption_high
	}
}

corruption_sumval = {
	type = country
	text = {
		trigger = {
			corruption_sum >= 0.1
		}
		localization_key = corruption_gain
	}
	text = {
		trigger = {
			corruption_sum <= 0.1
		}
		localization_key = corruption_loss
	}
}
