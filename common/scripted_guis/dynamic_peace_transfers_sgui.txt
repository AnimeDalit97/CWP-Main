	scope = country
	ai_is_valid = {
		#has_global_variable = china_dynamic_peace_signed
		#scope:country ?= c:SWI
		#always = yes
	}
	is_shown = {
		#has_global_variable = china_dynamic_peace_signed
		#scope:country ?= c:SWI
		#OR = {
		#	scope:country ?= c:PRC
		#	scope:country ?= c:ROC
		#}
	}

		#has_global_variable = china_dynamic_peace_signed
		#scope:country ?= c:SWI
	}
	saved_scopes = {
		owner
		controller
		state
		country
    }
	effect = {
		every_in_global_list = {
			variable = china_states
			if = {
				limit = {
					this ?= scope:state

					owner ?= c:ROC
				}
				set_variable = process_done
				#set_state_owner = scope:controller
				add_to_global_variable_list = { name = china_scripted_peace_PRC target = this }
			}
			if = {
				limit = {
					this ?= scope:state
					NOT = { scope:controller ?= c:PRC }
					NOT = { has_variable = process_done }
					owner ?= c:PRC
				}
				set_variable = process_done
				#set_state_owner = scope:controller
				add_to_global_variable_list = { name = china_scripted_peace_ROC target = this }
			}
		}
		#effect_china_states_scopes_transfer = yes
	}
}
