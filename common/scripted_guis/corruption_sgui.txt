showmodifier_corruption = {
    saved_scopes = { saved_gdp }
    effect = {
        random_country = {
            limit = {
                gdp = scope:saved_gdp
            }
			add_modifier = {
				name = corruption_modifier_effects
				multiplier = corruption_modifier
			}
        }
    }
}

showamount_corruption = {
	saved_scopes = { saved_gdp }
	effect = {
        random_country = {
            limit = {
                gdp = scope:saved_gdp
            }
			add = var:var_corruption
        }
    }
}

corruption_value = {
    effect = {
        every_country = {
            limit = {
                gdp = scope:saved_gdp
            }
			corruption_value_modifier = corruption_value
        }
    }
}
