		}
		if = {

			}
			add = {
				value = 50
				desc = "ideology_leftist"
			}
		}
		if = {
			limit = {

						has_law = law_decentralized

						has_law = law_council_republic

				scope:number = 3

			multiply = {
				value = 0.25
				desc = "from_three_prior_members"
			}
		}

			limit = {
				scope:number > 3
			}
			multiply = {
				value = 0.1
				desc = "from_more_than_three_prior_members"
