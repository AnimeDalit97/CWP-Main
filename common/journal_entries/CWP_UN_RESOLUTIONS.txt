# triggered by UN Resolution Screen
je_declaration_of_human_rights = {
	icon = "gfx/interface/icons/cwp_icons/united_nations.dds"

	possible = {
		OR = {
			has_variable = un_member
			has_variable = unsc_member
		}
	}

	on_weekly_pulse = {
		effect = {
			change_global_variable = {
				name = pending_resolution_progress
				add = 0.1
			}
		}
	}

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
		custom_tooltip = {
			text = "WILLCOMPLETEUN"
			un_support_value > un_oppose_value
		}
	}

	fail = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
		custom_tooltip = {
			text = "WILLFAILUN"
			un_oppose_value > un_support_value
		}
	}

	on_complete = {
		trigger_event = declaration_of_human_rights.1
		remove_global_variable = pending_resolution
		effect_reset_un_lists = yes
	}

	on_fail = {
		trigger_event = declaration_of_human_rights.2
		remove_global_variable = pending_resolution
		effect_reset_un_lists = yes
	}

	current_value = {
		value = global_var:pending_resolution_progress
	}

	goal_add_value = {
		add = 100
	}

	progressbar = yes

	weight = 100
}
